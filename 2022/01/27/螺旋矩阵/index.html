54. 螺旋矩阵

由于56题K神写法只适用于正方形，因此当题目变为矩形时，需要给后面的两个方向（右到左，下到上)增加判断条件，因为如果是正方形，最后到达中心点的时候，最后一次循环，必然是l=r,t=b,然后后两个方向就不会执行了，因为下一步就是r<l,t<b,但如果是矩形，那么就有可能出现l+1=r或者t+1=b在最后一次循环里，那么这时候就会导致最后两个方向会执行其中一个，导致最后结果多出一个值，这时候只需要加一个判断条件，如果返回的矩阵已经达到要求的数量，那么就不执行后面两个方向，强制不让后面两个方向执行即可。
/**

@param {number[][]} matrix
@return {number[]}
/
var spiralOrder = function(matrix) {
let n = matrix.length,m = matrix[0].length,l=t=0,r=m-1,b=n-1,mat= new Array(nm).fill(0),num=0,tar=nm
while(num<tar){
for(let i=l;i<=r;i++) mat[num++] = matrix[t][i]
t++
for(let i=t;i<=b;i++) mat[num++] = matrix[i][r]
r–
if(mat[nm-1]===0)for(let i=r;i>=l;i–) mat[num++] = matrix[b][i]
b–
if(mat[n*m-1]===0)for(let i=b;i>=t;i–) mat[num++] = matrix[i][l]
l++
}
return mat
};
